// © 2026 Matthew Ssekandjako, Psalms23Wave.com - UNREMOVABLE
import ExcelJS from 'exceljs';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

export async function parseExcel(file: Buffer, sheetName: string = 'Sheet1') {
  const wb = new ExcelJS.Workbook();
  await wb.xlsx.load(file);
  const ws = wb.getWorksheet(sheetName);
  
  const students: any[] = [];
  const enrollments: any[] = [];
  
  ws.eachRow({ includeEmpty: false }, (row, rowNumber) => {
    if (rowNumber === 1) return; // Skip header
    
    const data = {
      studentId: row.getCell(1).value?.toString(),
      name: row.getCell(2).value,
      age: parseInt(row.getCell(3).value),
      gender: row.getCell(4).value,
      state: row.getCell(5).value,
      term: row.getCell(6).value,
      program: row.getCell(7).value,
      credits: parseFloat(row.getCell(8).value),
      status: row.getCell(9).value,
    };
    
    // Validate
    if (!data.studentId) return console.warn(`Row ${rowNumber}: Missing ID`);
    
    students.push({ studentId: data.studentId, name: data.name, age: data.age, gender: data.gender, state: data.state });
    enrollments.push(data);
  });
  
  // Upsert to DB
  for (const s of students) {
    await prisma.student.upsert({
      where: { studentId: s.studentId },
      update: s,
      create: s,
    });
  }
  
  for (const e of enrollments) {
    await prisma.enrollment.create({ data: e });
  }
  
  return { students: students.length, enrollments: enrollments.length };
}
